Enums (Enumeration) - enum is a special data types that allows you to store a set of named constants. ðŸ‘‰ It makes code more readable instead of using numbers or strings directly.

enum Days
{
    Sunday,
    Monday,
    Tuesday,
    Wednesday,
    Thursday,
    Friday,
    Saturday
}

class Program
{
    static void Main()
    {
        Days today = Days.Monday;
        Console.WriteLine(today);          // Output: Monday
        Console.WriteLine((int)today);     // Output: 1 (default starts from 0)
    }
}

enum Days
{
    Sunday = 1,
    Monday = 2,
    Tuesday = 3
}

using System;

namespace MyApplication
{
  enum Level
  {
    Low,
    Medium,
    High
  }
  class Program
  {
    static void Main(string[] args)
    {
      Level myVar = Level.Medium;
      Console.WriteLine(myVar);
    }
  }
}
